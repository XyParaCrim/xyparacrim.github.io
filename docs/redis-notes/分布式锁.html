import java.util.HashMap;
import java.util.Map;
import java.util.Objects;

abstract class RedisReentrantLock {

   private static final ThreadLocal<Map<String, Integer>> localCounter = ThreadLocal.withInitial(HashMap::new);


   public boolean lock(String key) {
       Map<String, Integer> counter = localCounter.get();

       return counter.compute(key,
               (k, i) ->
                       i == null ?
                               tryAcquire(key) ?
                                       1 :
                                       null :
                               ++i
               ) != null;
   }

   public boolean unlock(String key) {
       Map<String, Integer> counter = localCounter.get();
       Integer count = counter.get(key);

       if (count == null) {
           return false;
       }

       if (count > 1) {
           counter.put(key, --count);
           return true;
       }


       return tryRelease(key) && Objects.equals(counter.remove(key), count);

   }
