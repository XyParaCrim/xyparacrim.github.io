import lombok.RequiredArgsConstructor;
import redis.clients.jedis.Jedis;
import redis.clients.jedis.Pipeline;
import redis.clients.jedis.Response;

import java.io.IOException;

@RequiredArgsConstructor
public class RedisLimiter {

    private final Jedis jedis;

    public boolean canExecute(String requestId, String action, int period, int maxCount) throws IOException {
        String key = normalizeKey(requestId, action);
        long now = System.currentTimeMillis();
        Pipeline pipeline = new Pipeline();

        pipeline.multi();
        pipeline.zadd(key, now, now + "");
        pipeline.zremrangeByScore(key, 0, now - period * 1000L);

        Response<Long> count = pipeline.zcard(key);

        pipeline.expire(key, period + 1);

        pipeline.exec();
        pipeline.close();

        return count.get() <= maxCount;
    }


    private static String normalizeKey(String requestId, String action) {
        return String.format("limit:%s:%s", requestId, action);
    }

}

import lombok.RequiredArgsConstructor;

import java.util.HashMap;
import java.util.Map;

@RequiredArgsConstructor
public class RedisLimiter {

    private final Map<String, Funnel> funnels = new HashMap<>();

    public boolean canExecute(String requestId, String action, int capacity, float leakingRate) {
        String key = normalizeKey(requestId, action);

        return funnels.
                computeIfAbsent(key, k -> new Funnel(capacity, leakingRate)).
                watering(1);
    }


    private static class Funnel {

        private final int capacity;

        private final float leakingRate;

        private int leftQuota;

        private long leakingTs;

        public Funnel(int capacity, float leakingRate) {
            this.capacity = capacity;
            this.leakingRate = leakingRate;
            this.leftQuota = capacity;
            this.leakingTs = System.currentTimeMillis();
        }

        void makeSpace() {
            long now = System.currentTimeMillis();
            long delta = now - leakingTs;
            int deltaQuota = (int) (delta * leakingRate);

            if (deltaQuota < 0) {
                tooLong();
                return;
            }

            if (deltaQuota < 1) {
                return;
            }

            leftQuota += deltaQuota;
            leakingTs = now;
            if (leftQuota > capacity) {
                leftQuota = capacity;
            }
        }

        void tooLong() {
            leftQuota = capacity;
            leakingTs = System.currentTimeMillis();
        }

        boolean watering(int quota) {
            makeSpace();
            if (leftQuota >= quota) {
                leftQuota -= quota;
                return true;
            }

            return false;
        }
    }

    private static String normalizeKey(String requestId, String action) {
        return String.format("limit:%s:%s", requestId, action);
    }


}
