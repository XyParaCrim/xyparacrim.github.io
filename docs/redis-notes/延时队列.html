import com.alibaba.fastjson.JSON;
import com.alibaba.fastjson.JSONObject;
import lombok.RequiredArgsConstructor;
import redis.clients.jedis.Jedis;

import java.util.Set;
import java.util.UUID;
import java.util.concurrent.locks.LockSupport;
import java.util.function.Consumer;

@RequiredArgsConstructor
public class RedisDelayQueue<T> implements Runnable {

    @RequiredArgsConstructor
    private static class Node<T> {

        private final String id = UUID.randomUUID().toString();

        private final T value;

    }


    private final Jedis jedis;

    private final Consumer<T> handler;

    private final String key = UUID.randomUUID().toString();


    public void add(T value) {
        jedis.zadd(
                key,
                System.currentTimeMillis() + 5000,
                JSON.toJSONString(new Node<>(value))
        );
    }

    @Override
    public void run() {
        while (!Thread.currentThread().isInterrupted()) {
            Set<String> values = jedis.zrangeByScore(key, 0, System.currentTimeMillis(), 0, 1);

            if (values.isEmpty()) {
                LockSupport.parkUntil(500);
                continue;
            }

            String s = values.stream().findFirst().get();
            if (jedis.zrem(key, s) > 0) {
                @SuppressWarnings("unchecked")
                Node<T> node = (Node<T>) JSONObject.parseObject(s, Node.class);

                handler.accept(node.value);
            }
        }
    }


}
